#include <stdio.h>
#include <stdlib.h>

{

gets(a);

/*Ciphering*/
for(i=0,j=0;i<l;i++)
{
if(i%2==0)c[j++]=a[i];
}
for(i=0;i<l;i++)
{
if(i%2==1)c[j++]=a[i];
}
c[j]='\0';
printf("\nCiphertextafterapplyingrailfence:");
printf("\n%s",c);
/*Deciphering*/
if(i%2==0)
k=l/2;
else
k=(l/2)+1;
for(i=0,j=0;i<k;i++)
{
d[j]=c[i];j=j+2;
}
for(i=k,j=1;i<l;i++)
{
d[j]=c[i];
j=j+2;
}

}
